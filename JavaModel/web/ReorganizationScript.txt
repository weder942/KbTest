CREATE SEQUENCE TGitTestId MINVALUE 1 INCREMENT BY 1
CREATE TABLE TGitTest (TGitTestId NUMERIC(4) NOT NULL , TGitTestDesc VARCHAR2(40) NOT NULL , PRIMARY KEY(TGitTestId))
CREATE OR REPLACE TRIGGER AN$1TGitTestId BEFORE INSERT ON TGitTest FOR EACH ROW BEGIN SELECT TGitTestId.NEXTVAL INTO :new.TGitTestId FROM DUAL; END;
